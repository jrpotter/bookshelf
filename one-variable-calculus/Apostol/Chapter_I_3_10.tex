\begin{xtheorem}{1.30}[Archimedean Property of the Reals]

\begin{xtheorem}{1.31}

If three real numbers $a$, $x$, and $y$ satisfy the inequalities
$$a \leq x \leq a + \frac{y}{n}$$
for every integer $n \geq 1$, then $x = a$.

\end{xtheorem}

\begin{proof}
